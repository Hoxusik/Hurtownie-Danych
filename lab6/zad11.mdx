SELECT 
NON EMPTY { [Measures].[Wynik z Matury] } ON COLUMNS, 
NON EMPTY { 
    GENERATE(
        EXCEPT(
            [ID Daty Matury].[Rok].[Rok].ALLMEMBERS, 
            [ID Daty Matury].[Rok].[Rok].[Unknown]
        ),
        TOPCOUNT(
            [ID Daty Matury].[Rok].CurrentMember * [Dim Przedmiot].[Nazwa].[Nazwa].ALLMEMBERS *
			[Dim Uczen].[Imie Nazwisko].[Imie Nazwisko].ALLMEMBERS, 
            1, 
            [Measures].[Wynik z Matury]
        )
    )
} ON ROWS 

FROM [Szkola30] 

WHERE ([Dim Junk].[Ocena].&[2])
